# Anlisis de Matriz de Confusin para el Modelo Naive Bayes
import pandas as pd
import numpy as np
import matplotlib.pyplot as plt
import seaborn as sns
from sklearn.metrics import confusion_matrix

# Cargar y preparar los datos (asumo que ya tienes estos datos disponibles)
# X_test, y_test, y_pred deben estar disponibles

# Crear la matriz de confusin
cm = confusion_matrix(y_test, y_pred)



# Calcular mtricas por clase
clases = ['Econmica', 'Intermedia', 'Cara']
precision_por_clase = []
recall_por_clase = []
f1_por_clase = []

total_casos = np.sum(cm)
aciertos_totales = np.sum(np.diag(cm))
accuracy_global = aciertos_totales / total_casos


for i, clase in enumerate(clases):
    # Verdaderos positivos
    tp = cm[i, i]
    
    # Total de casos reales de esta clase
    total_real = np.sum(cm[i, :])
    
    # Total de predicciones para esta clase
    total_pred = np.sum(cm[:, i])
    
    # Precisin: De todos los que predije como clase i, cuntos realmente lo eran?
    precision = tp / total_pred if total_pred > 0 else 0
    
    # Recall: De todos los que realmente son clase i, cuntos identifiqu correctamente?
    recall = tp / total_real if total_real > 0 else 0
    
    # F1-Score: Media armnica de precisin y recall
    f1 = 2 * (precision * recall) / (precision + recall) if (precision + recall) > 0 else 0
    
    precision_por_clase.append(precision)
    recall_por_clase.append(recall)
    f1_por_clase.append(f1)
    
    print(f"{clase:<12}{precision:.4f}{recall:.4f}{f1:.4f}{tp:<12}{total_real:<12}{total_pred:<12}")

# Anlisis de errores
print("\nAnlisis de errores:")
errores_totales = total_casos - aciertos_totales
print(f"Total de errores: {errores_totales} de {total_casos} casos ({errores_totales/total_casos:.4f})")

# Analizar los tipos especficos de errores
