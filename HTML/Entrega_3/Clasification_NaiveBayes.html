# Análisis de Matriz de Confusión para el Modelo Naive Bayes
import pandas as pd
import numpy as np
import matplotlib.pyplot as plt
import seaborn as sns
from sklearn.metrics import confusion_matrix

# Cargar y preparar los datos (asumo que ya tienes estos datos disponibles)
# X_test, y_test, y_pred deben estar disponibles

# Crear la matriz de confusión
cm = confusion_matrix(y_test, y_pred)



# Calcular métricas por clase
clases = ['Económica', 'Intermedia', 'Cara']
precision_por_clase = []
recall_por_clase = []
f1_por_clase = []

total_casos = np.sum(cm)
aciertos_totales = np.sum(np.diag(cm))
accuracy_global = aciertos_totales / total_casos


for i, clase in enumerate(clases):
    # Verdaderos positivos
    tp = cm[i, i]
    
    # Total de casos reales de esta clase
    total_real = np.sum(cm[i, :])
    
    # Total de predicciones para esta clase
    total_pred = np.sum(cm[:, i])
    
    # Precisión: De todos los que predije como clase i, ¿cuántos realmente lo eran?
    precision = tp / total_pred if total_pred > 0 else 0
    
    # Recall: De todos los que realmente son clase i, ¿cuántos identifiqué correctamente?
    recall = tp / total_real if total_real > 0 else 0
    
    # F1-Score: Media armónica de precisión y recall
    f1 = 2 * (precision * recall) / (precision + recall) if (precision + recall) > 0 else 0
    
    precision_por_clase.append(precision)
    recall_por_clase.append(recall)
    f1_por_clase.append(f1)
    
    print(f"{clase:<12}{precision:.4f}{recall:.4f}{f1:.4f}{tp:<12}{total_real:<12}{total_pred:<12}")

# Análisis de errores
print("\nAnálisis de errores:")
errores_totales = total_casos - aciertos_totales
print(f"Total de errores: {errores_totales} de {total_casos} casos ({errores_totales/total_casos:.4f})")

# Analizar los tipos específicos de errores
print("\nTipos específicos de errores:")
tipos_error = []

# Error Tipo 1: Económica clasificada como Intermedia
tipos_error.append(("Económica → Intermedia", cm[0, 1]))

# Error Tipo 2: Económica clasificada como Cara
tipos_error.append(("Económica → Cara", cm[0, 2]))

# Error Tipo 3: Intermedia clasificada como Económica
tipos_error.append(("Intermedia → Económica", cm[1, 0]))

# Error Tipo 4: Intermedia clasificada como Cara
tipos_error.append(("Intermedia → Cara", cm[1, 2]))

# Error Tipo 5: Cara clasificada como Económica
tipos_error.append(("Cara → Económica", cm[2, 0]))

# Error Tipo 6: Cara clasificada como Intermedia
tipos_error.append(("Cara → Intermedia", cm[2, 1]))

# Ordenar los errores por frecuencia, de mayor a menor
tipos_error.sort(key=lambda x: x[1], reverse=True)

for tipo, cantidad in tipos_error:
    if cantidad > 0:
        print(f"{tipo}: {cantidad} casos ({cantidad/errores_totales:.4f} de los errores)")

# Análisis de la gravedad de los errores
print("\nGravedad de los errores:")
errores_graves = cm[0, 2] + cm[2, 0]  # Económica como Cara o viceversa
errores_leves = cm[0, 1] + cm[1, 0] + cm[1, 2] + cm[2, 1]  # Errores entre categorías adyacentes

print(f"Errores graves (salto de 2 categorías): {errores_graves} casos ({errores_graves/errores_totales:.4f} de los errores)")
print(f"Errores leves (salto de 1 categoría): {errores_leves} casos ({errores_leves/errores_totales:.4f} de los errores)")
