PREDICCIN DE PRECIOS DE CASAS CON REDES NEURONALES

Este script utiliza redes neuronales para predecir el precio de casas basado en varias caractersticas como calidad general, rea habitable, capacidad del garaje, entre otras.

Descripcin:

Este programa implementa dos modelos de redes neuronales con diferentes arquitecturas para predecir precios de casas. Utiliza un conjunto de caractersticas seleccionadas para entrenar los modelos y evala su rendimiento mediante mtricas de regresin. Tambin incluye ejemplos de prediccin para casos especficos.

Caractersticas utilizadas:

  • OverallQual: Calidad general de la casa (1-10)
  • GrLivArea: rea habitable en pies cuadrados
  • GarageCars: Capacidad del garaje en nmero de autos
  • YearBuilt: Ao de construccin
  • TotalBsmtSF: Superficie total del stano en pies cuadrados
  • FullBath: Nmero de baos completos
  • Neighborhood: Vecindario (categrica)
  • ExterQual: Calidad del exterior (categrica)
  • KitchenQual: Calidad de la cocina (categrica)
  • BsmtQual: Calidad del stano (categrica)

def evaluate_model(model, X_train, y_train, X_test, y_test, model_name):
    """
    Evala el rendimiento del modelo en los conjuntos de entrenamiento y prueba.
    
    Parmetros:
    -----------
    model : Pipeline
        Modelo de red neuronal entrenado
    X_train : DataFrame
        Caractersticas del conjunto de entrenamiento
    y_train : Series
        Variable objetivo del conjunto de entrenamiento
    X_test : DataFrame
        Caractersticas del conjunto de prueba
    y_test : Series
        Variable objetivo del conjunto de prueba
    model_name : str
        Nombre del modelo para mostrar en los resultados
        
    Retorna:
    --------
    tuple : (RMSE en prueba, R en prueba)
    """
    # PASO 1: Obtenemos predicciones en el conjunto de entrenamiento
    y_train_pred = model.predict(X_train)
    train_mse = mean_squared_error(y_train, y_train_pred)
    train_rmse = np.sqrt(train_mse)
    train_r2 = r2_score(y_train, y_train_pred)
    
    # PASO 2: Obtenemos predicciones en el conjunto de prueba
    y_test_pred = model.predict(X_test)
    test_mse = mean_squared_error(y_test, y_test_pred)
    test_rmse = np.sqrt(test_mse)
    test_r2 = r2_score(y_test, y_test_pred)
    
    # PASO 3: Mostramos mtricas de regresin
    print(f"\n{model_name} - Mtricas de Regresin:")
    print(f"RMSE en Entrenamiento: ${train_rmse:.2f}")
    print(f"R en Entrenamiento: {train_r2:.4f}")
    print(f"RMSE en Prueba: ${test_rmse:.2f}")
    print(f"R en Prueba: {test_r2:.4f}")
    
    # PASO 4: Calculamos mtricas de clasificacin (binning de precios)
    try:
        # Limpiamos valores NaN
        y_train_clean = pd.Series(y_train).fillna(y_train.median())
        y_test_clean = pd.Series(y_test).fillna(y_test.median())
        y_train_pred_clean = pd.Series(y_train_pred).fillna(np.median(y_train_pred))
        y_test_pred_clean = pd.Series(y_test_pred).fillna(np.median(y_test_pred))
        
        # Creamos bins basados en cuartiles de la variable objetivo
        all_values = pd.concat([y_train_clean, y_test_clean])
        bins = [
            all_values.min() - 1,           # Mnimo con margen
            all_values.quantile(0.25),      # Primer cuartil (25%)
            all_values.quantile(0.5),       # Mediana (50%)
            all_values.quantile(0.75),      # Tercer cuartil (75%)
            all_values.max() + 1            # Mximo con margen
        ]
        labels = [0, 1, 2, 3]  # 4 categoras de precio
        
        # Convertimos valores continuos a categoras
        y_train_binned = pd.cut(y_train_clean, bins=bins, labels=labels).astype(int)
        y_test_binned = pd.cut(y_test_clean, bins=bins, labels=labels).astype(int)
        
        # Aseguramos que los valores predichos estn dentro del rango
        y_train_pred_clean = np.clip(y_train_pred_clean, all_values.min(), all_values.max())
        y_test_pred_clean = np.clip(y_test_pred_clean, all_values.min(), all_values.max())
        
        # Discretizamos las predicciones
        y_train_pred_binned = pd.cut(y_train_pred_clean, bins=bins, labels=labels).astype(int)
        y_test_pred_binned = pd.cut(y_test_pred_clean, bins=bins, labels=labels).astype(int)
        
        # Calculamos mtricas de clasificacin
        cm = confusion_matrix(y_test_binned, y_test_pred_binned)
        accuracy = accuracy_score(y_test_binned, y_test_pred_binned)
        precision = precision_score(y_test_binned, y_test_pred_binned, average='micro')
        recall = recall_score(y_test_binned, y_test_pred_binned, average='micro')
        f1 = f1_score(y_test_binned, y_test_pred_binned, average='micro')
        
        # Mostramos mtricas de clasificacin
        print(f"\n{model_name} - Mtricas de Clasificacin (despus de discretizar precios):")
        print('Matriz de Confusin:\n', cm)
        print('Exactitud (Accuracy): ', accuracy)
        print('Precisin (Precision): ', precision)
        print('Exhaustividad (Recall): ', recall)
        print('Puntuacin F1 (F1 Score): ', f1)
    except Exception as e:
        print(f"\nError al calcular mtricas de clasificacin: {e}")
        print("Continuando con mtricas de regresin y visualizaciones...")
    
    # PASO 5: Visualizacin - Precios reales vs predichos
    plt.figure(figsize=(10, 6))
    plt.scatter(y_test, y_test_pred, alpha=0.5)
    plt.plot([y_test.min(), y_test.max()], [y_test.min(), y_test.max()], 'r--')
    plt.xlabel('Precio Real')
    plt.ylabel('Precio Predicho')
    plt.title(f'{model_name}: Precios Reales vs Predichos')
    plt.show()
    
    # PASO 6: Visualizacin - Grfico de residuos
    residuals = y_test - y_test_pred
    plt.figure(figsize=(10, 6))
    plt.scatter(y_test_pred, residuals, alpha=0.5)
    plt.hlines(y=0, xmin=y_test_pred.min(), xmax=y_test_pred.max(), colors='r', linestyles='--')
    plt.xlabel('Precio Predicho')
    plt.ylabel('Residuos')
    plt.title(f'{model_name}: Grfico de Residuos')
    plt.show()
    
    return test_rmse, test_r2
