# Cargar los datos
df = pd.read_csv('./train.csv')

# Calcular los tercilos (dividir los datos en tres partes iguales)
tercil_1 = np.percentile(df['SalePrice'], 33.33)
tercil_2 = np.percentile(df['SalePrice'], 66.67)

print(f"Tercil 1 (33.33%): ${int(tercil_1):,}")
print(f"Tercil 2 (66.67%): ${int(tercil_2):,}")

# 1. Crear la variable de clasificacin
df['PriceCategory'] = pd.cut(
    df['SalePrice'],
    bins=[0, tercil_1, tercil_2, float('inf')],
    labels=['Econmico', 'Medio', 'Premium'],
    include_lowest=True
)

# Verificar la distribucin de las categoras
category_counts = df['PriceCategory'].value_counts().sort_index()
print("\nDistribucin de categoras:")
print(category_counts)
print(f"Porcentajes: {category_counts / len(df) * 100}")

# 2. Seleccionar las caractersticas para el modelo
# Basado en nuestro anlisis previo, seleccionamos las caractersticas ms importantes
selected_features = [
    'OverallQual',    # Calidad general
    'GrLivArea',      # rea habitable
    'GarageCars',     # Capacidad del garaje
    'YearBuilt',      # Ao de construccin
    'TotalBsmtSF',    # Superficie total del stano
    'FullBath',       # Baos completos
    'Neighborhood',   # Vecindario (categrica)
    'ExterQual',      # Calidad del exterior (categrica)
    'KitchenQual',    # Calidad de la cocina (categrica)
    'BsmtQual'        # Calidad del stano (categrica)
]

# 3. Preparar las caractersticas para el modelo
X = df[selected_features].copy()
y = df['PriceCategory']

# Manejar valores faltantes en caractersticas numricas
numeric_features = X.select_dtypes(include=['int64', 'float64']).columns
imputer = SimpleImputer(strategy='median')
X[numeric_features] = imputer.fit_transform(X[numeric_features])

# Manejar valores faltantes en caractersticas categricas
categorical_features = X.select_dtypes(include=['object']).columns
X[categorical_features] = X[categorical_features].fillna('None')

# Codificar variables categricas
encoders = {}
for feature in categorical_features:
    le = LabelEncoder()
    X[feature] = le.fit_transform(X[feature])
    encoders[feature] = le


# 4. Crear y entrenar el modelo Random Forest
rf_classifier = RandomForestClassifier(
    n_estimators=1000,        # Nmero de rboles en el bosque
    max_depth=7,             # Profundidad mxima de los rboles
    min_samples_split=20,    # Mnimo de muestras para dividir un nodo
    min_samples_leaf=10,     # Mnimo de muestras en un nodo hoja
    random_state=42,
    n_jobs=-1                # Usar todos los ncleos disponibles para mayor rendimiento
)

X_train, X_test, y_train, y_test = train_test_split(
    X, y, test_size=0.3, random_state=42, stratify=y
)

rf_classifier.fit(X_train, y_train)

# 5. Evaluar el modelo
y_pred = rf_classifier.predict(X_test)

# ------- ANLISIS DE EFICIENCIA DEL MODELO -------
# Para modelos de clasificacin, adaptamos las mtricas de regresin
# Convertir categoras a valores numricos para calcular R2, MAE, MSE
label_map = {'Econmico': 0, 'Medio': 1, 'Premium': 2}
y_test_numeric = y_test.map(label_map)
y_pred_numeric = np.array([label_map[pred] for pred in y_pred])

conf_matrix = confusion_matrix(y_test, y_pred)

print("\nReporte de clasificacin:")
print(classification_report(y_test, y_pred))

# Calcular R2, MAE y MSE
r2 = r2_score(y_test_numeric, y_pred_numeric)
mae = mean_absolute_error(y_test_numeric, y_pred_numeric)
mse = mean_squared_error(y_test_numeric, y_pred_numeric)
rmse = np.sqrt(mse)

accuracy = accuracy_score(y_test, y_pred)
# Crear un dataframe con un resumen de las mtricas
metrics_summary = pd.DataFrame({
    'Mtrica': ['Accuracy', 'R', 'MAE', 'MSE', 'RMSE'],
    'Valor': [accuracy, r2, mae, mse, rmse]
})

print("\n--- RESUMEN DE MTRICAS DE EFICIENCIA ---")
print(metrics_summary)
