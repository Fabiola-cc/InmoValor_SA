# Definir el modelo KNN para regresi√≥n
knn_regressor = KNeighborsRegressor()

# Configurar validaci√≥n cruzada (k-fold)
k_folds = 5
kf = KFold(n_splits=k_folds, shuffle=True, random_state=42)

# Realizar validaci√≥n cruzada para evaluar diferentes valores de k (n√∫mero de vecinos)
k_range = list(range(1, 31))
k_scores = []

for k in k_range:
    knn = KNeighborsRegressor(n_neighbors=k)
    # Usamos el negativo del error cuadr√°tico medio como m√©trica
    scores = cross_val_score(knn, X_train, y_train, cv=kf, scoring='neg_mean_squared_error')
    k_scores.append(-scores.mean())  # Convertimos a positivo para facilitar la interpretaci√≥n

# Visualizar los resultados de la validaci√≥n cruzada
plt.figure(figsize=(10, 6))
plt.plot(k_range, k_scores)
plt.xlabel('Valor de K')
plt.ylabel('Error Cuadr√°tico Medio (MSE)')
plt.title('Error de Validaci√≥n Cruzada vs. Valor de K')
plt.xticks(k_range[::2])  # Mostrar cada segundo valor en el eje X
plt.grid(True)
plt.show()

# Encontrar el mejor valor de k
best_k = k_range[np.argmin(k_scores)]
print(f"Mejor valor de k: {best_k}")

Mejor valor de k: 2

# Definir la cuadr√≠cula de par√°metros
param_grid = {
    'n_neighbors': list(range(1, 21)),
    'weights': ['uniform', 'distance'],
    'p': [1, 2]  # p=1 para distancia Manhattan, p=2 para distancia Euclidiana
}

# Configurar GridSearchCV
grid_search = GridSearchCV(
    KNeighborsRegressor(),
    param_grid,
    cv=5,
    scoring='neg_mean_squared_error',
    n_jobs=-1  # Usar todos los n√∫cleos disponibles
)

# Ajustar el modelo
grid_search.fit(X_train, y_train)

# Imprimir los mejores par√°metros
print("Mejores par√°metros encontrados:")

# Evaluaci√≥n con validaci√≥n cruzada
cv_scores = cross_val_score(best_knn, X, y, cv=5, scoring='neg_mean_squared_error')
rmse_cv = np.sqrt(-cv_scores)

print("Resultados de la validaci√≥n cruzada (RMSE):")
for i, rmse in enumerate(rmse_cv):
    print(f"Fold {i+1}: ${int(rmse):,}")
print(f"RMSE promedio: ${int(rmse_cv.mean()):,}")
print(f"Desviaci√≥n est√°ndar: ${int(rmse_cv.std()):,}")

# Predecir en el conjunto de prueba con el mejor modelo
y_pred = best_knn.predict(X_test)

# Calcular m√©tricas de evaluaci√≥n
mse = mean_squared_error(y_test, y_pred)
rmse = np.sqrt(mse)
mae = mean_absolute_error(y_test, y_pred)
r2 = r2_score(y_test, y_pred)

print("\nResultados en el conjunto de prueba:")
print(f"Error cuadr√°tico medio (MSE): ${int(mse):,}")
print(f"Ra√≠z del error cuadr√°tico medio (RMSE): ${int(rmse):,}")
print(f"Error absoluto medio (MAE): ${int(mae):,}")
print(f"Coeficiente de determinaci√≥n (R¬≤): {r2:.4f}")
